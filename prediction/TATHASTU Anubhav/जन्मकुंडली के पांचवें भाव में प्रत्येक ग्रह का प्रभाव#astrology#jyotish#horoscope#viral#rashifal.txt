/**
 * Main function to generate predictions, structured house by house.
 * @param {Array} chartData - The chartx array. [{id, long, sign, bhava, retro},...]
 * @returns {object} - Object with 'hindi' and 'english' HTML prediction strings.
 */
function generatePredictions(chartData) {
    let predictionHindi = "=== ज्योतिषीय विश्लेषण ===<br><br>";
    let predictionEnglish = "=== Astrological Analysis ===<br><br>";
    let remediesHindi = "<br><br>=== संभावित उपाय ===";
    let remediesEnglish = "<br><br>=== Potential Remedies ===";
    let remedyAdded = false; // Flag to track if any remedy was added

    // --- Validations ---
    if (!Array.isArray(chartData) || chartData.length === 0) {
        return { hindi: "<p>त्रुटि: अमान्य या खाली चार्ट डेटा।</p>", english: "<p>Error: Invalid or empty chart data.</p>" };
    }
    // Check if essential data structures are loaded
    if (typeof PLANET_IDS === 'undefined' || typeof planetNames === 'undefined' || typeof signNames === 'undefined' || typeof houseNames === 'undefined' || typeof dignityInfo === 'undefined' || typeof nakshatras === 'undefined' || typeof houseKarakas === 'undefined' || typeof planetRemedies === 'undefined' || typeof vimsottariDasaYears === 'undefined' || typeof subLordSpans === 'undefined' || typeof vimsottariSequence === 'undefined') {
        return { hindi: "<p>त्रुटि: आवश्यक ज्योतिषीय डेटा लोड नहीं हुआ है।</p>", english: "<p>Error: Required astrological data is not loaded.</p>" };
    }

    const ascData = getPlanetDataById(chartData, PLANET_IDS.ASC);
    // Validate Ascendant data thoroughly
    if (!ascData || !ascData.sign || typeof ascData.sign !== 'number' || ascData.sign < 1 || ascData.sign > 12 || typeof ascData.long !== 'number' || isNaN(ascData.long)) {
         console.error("Invalid or incomplete Ascendant data:", ascData);
        return { hindi: "<p>त्रुटि: लग्न डेटा अपूर्ण या अमान्य है।</p>", english: "<p>Error: Ascendant data is incomplete or invalid.</p>" };
    }

    const ascSign = ascData.sign;
    const ascSignInfo = signNames?.[ascSign];
    // Validate Ascendant sign info and its lord
    if (!ascSignInfo || ascSignInfo.lord === undefined || ascSignInfo.lord === null) {
        console.error(`Invalid data for Ascendant sign (${ascSign}) or its lord.`, ascSignInfo);
        return { hindi: `<p>त्रुटि: लग्न राशि (${ascSign}) या उसके स्वामी के लिए डेटा अमान्य है।</p>`, english: `<p>Error: Data for Ascendant sign (${ascSign}) or its lord is invalid.</p>` };
    }

    const lagnaLordId = ascSignInfo.lord;
    const lagnaLordData = getPlanetDataById(chartData, lagnaLordId);
    const houseLordships = calculateHouseLordships(ascSign);
    const grahaYuddhas = checkGrahaYuddha(chartData);
    const allAspects = calculateAspects(chartData);
    const parivartanaYogas = checkParivartanaYoga(chartData, signNames);

    // --- Ascendant Details ---
    try {
        const lagnaLordNameHi = planetNames?.[lagnaLordId]?.hi ?? `ग्रह ${lagnaLordId}`;
        const lagnaLordShortHi = planetNames?.[lagnaLordId]?.short_hi ?? '?';
        const lagnaLordNameEn = planetNames?.[lagnaLordId]?.en ?? `Planet ${lagnaLordId}`;
        const lagnaLordShortEn = planetNames?.[lagnaLordId]?.short ?? '?';

        predictionHindi += `<b>लग्न / Ascendant (${planetNames?.[PLANET_IDS.ASC]?.short_hi ?? 'Lg'})</b><br>`;
        predictionEnglish += `<b>Ascendant / लग्न (${planetNames?.[PLANET_IDS.ASC]?.short ?? 'Asc'})</b><br>`;
        predictionHindi += `लग्न: ${ascSignInfo.hi} (${ascSignInfo.en}), स्वामी: ${lagnaLordNameHi} (${lagnaLordShortHi}). देशांतर: ${formatLongitude(ascData.long)}.`;
        predictionEnglish += `Ascendant: ${ascSignInfo.en} (${ascSignInfo.hi}), Lord: ${lagnaLordNameEn} (${lagnaLordShortEn}). Longitude: ${formatLongitude(ascData.long)}.`;

        const ascNakshatra = calculateNakshatra(ascData.long);
        const ascSubLord = calculateSubLord(ascData.long);
        if (ascNakshatra) {
            predictionHindi += ` नक्षत्र: ${ascNakshatra.nameHi} (स्वा: ${ascNakshatra.lordShortHi}).`;
            predictionEnglish += ` Nakshatra: ${ascNakshatra.name} (Lo: ${ascNakshatra.lordShortEn}).`;
        }
        if (ascSubLord) {
            predictionHindi += ` उप: ${ascSubLord.shortHi}.`;
            predictionEnglish += ` Sub: ${ascSubLord.shortEn}.`;
        }
        predictionHindi += '<br>';
        predictionEnglish += '<br>';

        // Ascendant Lord Placement Analysis

            } else {

                const llHouseEn = houseNames[llHouse].en.split('(')[0].trim();
                const llHouseHi = houseNames[llHouse].hi.split('(')[0].trim();
                const llSignEn = signNames[llSign].en;
                const llSignHi = signNames[llSign].hi;

                predictionHindi += `लग्नेश ${lagnaLordShortHi}, ${llHouseHi} में ${llSignHi} (${llDignity.hi}) में स्थित है। `;
                predictionEnglish += `Asc. Lord ${lagnaLordShortEn}, is placed in ${llHouseEn} in ${llSignEn} (${llDignity.en}). `;

                // Interpretation based on house placement (Leveraging "अकेला लग्नेश बना देता है राजयोग..." text)
                let pHi = ''; let pEn = '';
                const goodHousesForLagnesh = [1, 2, 4, 5, 7, 9, 10, 11]; // From the text

                const lessIdealHouses = [3];

                    pHi += `यह एक शुभ स्थिति है जो राजयोग के समान परिणाम दे सकती है, संबंधित भाव के क्षेत्रों में सफलता और मजबूती लाती है। `;
                    pEn += `This is an auspicious placement that can yield Rajayoga-like results, bringing success and strength in areas related to that house. `;
                }
                 else if (dusthanaHouses.includes(llHouse)) {
                     pHi += `त्रिक भाव (${llHouseHi}) में लग्नेश का होना कुछ चुनौतियाँ प्रस्तुत कर सकता है, विशेषकर स्वास्थ्य, बाधाओं या खर्चों से संबंधित। `;
                     pEn += `Ascendant Lord in a Dusthana house (${llHouseEn}) can present certain challenges, particularly related to health, obstacles, or expenses. `;

                          const remedyKeyHi = `\n* <b>${lagnaLordNameHi}`;

                            remediesEnglish += `\n* <b>${lagnaLordNameEn} (Asc Lord):</b> ${remedy.general_en}`;
                            remedyAdded = true;
                          }
                     }
                } else if (llHouse === 3) {
                     pHi += `तृतीय भाव (उपचय) में लग्नेश आपको साहसी और मेहनती बनाता है, लेकिन सफलता के लिए निरंतर प्रयास और संचार कौशल महत्वपूर्ण हैं। `;
                     pEn += `The Ascendant Lord in the 3rd house (Upachaya) makes you courageous and hardworking, but consistent effort and communication skills are key to success. `;
                }
                predictionHindi += pHi;
                predictionEnglish += pEn;
            }
        } else {
            predictionHindi += `लग्नेश (${lagnaLordNameHi}) का डेटा चार्ट में उपलब्ध नहीं है या अपूर्ण है।<br>`;
            predictionEnglish += `Ascendant Lord (${lagnaLordNameEn}) data is not available or incomplete in the chart.<br>`;
        }

         // Analysis based on placement of Bhagyesh relative to Lagnesh
         const bhagyeshId = houseLordships[9];
         if (bhagyeshId !== null && planetNames?.[bhagyeshId]) {
             const bhagyeshData = getPlanetDataById(chartData, bhagyeshId);
             const bhagyeshNameHi = planetNames[bhagyeshId].hi;
              const bhagyeshNameEn = planetNames[bhagyeshId].en;

             if (lagnaLordData && lagnaLordData.bhava && bhagyeshData && bhagyeshData.bhava) {
                 const diff = ((bhagyeshData.bhava - lagnaLordData.bhava + 12) % 12) + 1;
                 const houseFromLagnesh = diff;
                 let trikonaType = "";
                 let trikonaTypeHi = "";

                 if ([1, 5, 9].includes(houseFromLagnesh)) { trikonaType = "Dharma Trikona (Fortune)"; trikonaTypeHi = "धर्म त्रिकोण (भाग्य)"; }

                 else if ([3, 7, 11].includes(houseFromLagnesh)) { trikonaType = "Kama Trikona (Desire/Action)"; trikonaTypeHi = "काम त्रिकोण (इच्छा/क्रिया)"; }
                 else if ([4, 8, 12].includes(houseFromLagnesh)) { trikonaType = "Moksha Trikona (Liberation/Spirituality)"; trikonaTypeHi = "मोक्ष त्रिकोण (मुक्ति/अध्यात्म)"; }

                 if (trikonaType) {
                     predictionHindi += `<br>भाग्येश (${bhagyeshNameHi}) लग्नेश से ${houseFromLagnesh} घर आगे (${trikonaTypeHi}) में स्थित हैं। `;
                     predictionEnglish += `<br>The Lord of Fortune (${bhagyeshNameEn}) is placed ${houseFromLagnesh} houses ahead of the Ascendant Lord (in ${trikonaType}). `;
